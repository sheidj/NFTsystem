### 1. 毕业生注册功能

#### 1.1 单个注册流程

**用户操作路径**：管理员面板 → 单个注册标签 → 填写信息 → 点击注册

**完整数据流**：

```
1. 用户输入
   └→ frontend/app/admin/page.tsx (组件状态)
       ├─ graduateAddress: 钱包地址
       ├─ studentId: 学号
       ├─ studentName: 姓名
       ├─ selectedCollege: 学院ID
       └─ selectedMajor: 专业ID

2. 表单提交
   └→ handleRegister() 函数
       └→ walletClient.writeContract()
           ├─ address: contractAddress
           ├─ abi: contractABI
           ├─ functionName: 'registerGraduate'
           └─ args: [address, studentId, name, major, college]

3. 智能合约执行
   └→ contracts/GraduationNFT.sol
       └→ registerGraduate() 函数
           ├─ 验证地址有效性
           ├─ 验证学号唯一性
           ├─ 创建Graduate结构体
           ├─ 存储到graduates映射
           ├─ 建立studentIdToAddress映射
           ├─ 添加到graduateAddresses数组
           └─ 触发GraduateRegistered事件

4. 数据库同步
   └→ frontend/app/api/db/user/route.ts
       └→ POST处理函数
           ├─ 接收注册数据
           ├─ Prisma upsert操作
           └─ 返回用户记录
```

  ```solidity
  function registerGraduate(
      address graduateAddress,
      string memory studentId,
      string memory studentName,
      string memory major,
      string memory college
  ) external onlyRole(ADMIN_ROLE) {
      require(graduateAddress != address(0), "Invalid address");
      require(bytes(studentId).length > 0, "Student ID required");
      require(bytes(graduates[graduateAddress].studentId).length == 0, "Already registered");
      require(studentIdToAddress[studentId] == address(0), "Student ID already used");
      
      graduates[graduateAddress] = Graduate({
          studentId: studentId,
          studentName: studentName,
          major: major,
          college: college,
          mintedAt: 0,
          hasClaimed: false,
          selfMintCount: 0,
          pityCounter: 0
      });
      
      studentIdToAddress[studentId] = graduateAddress;
      graduateAddresses.push(graduateAddress);
      
      emit GraduateRegistered(graduateAddress, studentId, studentName);
  }
  ```

- **数据库同步**：`frontend/app/api/db/user/route.ts`
  ```typescript
  export async function POST(request: Request) {
    const body = await request.json();
    const { walletAddress, studentId, name, college, major, isRegistered } = body;
    
    const user = await prisma.user.upsert({
      where: { walletAddress: walletAddress.toLowerCase() },
      update: { studentId, name, college, major, isRegistered },
      create: { walletAddress: walletAddress.toLowerCase(), studentId, name, college, major, isRegistered },
    });
    
    return NextResponse.json(user);
  }
  ```

**批量注册格式**：

  ```typescript
  const handleBatchRegister = async () => {
    const validStudents = parsedStudents.filter(s => s.isValid);
    setBatchRegistering(true);
    setBatchRegisterProgress({ current: 0, total: validStudents.length });
    
    let success = 0;
    let failed = 0;
    
    for (let i = 0; i < validStudents.length; i++) {
      const student = validStudents[i];
      setBatchRegisterProgress({ current: i + 1, total: validStudents.length });
      
      try {
        // 检查是否已注册
        const isRegistered = await publicClient.readContract({...});
        if (isRegistered) { failed++; continue; }
        
        // 注册
        const hash = await walletClient.writeContract({
          functionName: 'registerGraduate',
          args: [student.address, student.studentId, ...],
        });
        
        await publicClient.waitForTransactionReceipt({ hash });
        success++;
      } catch (error) {
        failed++;
      }
    }
    
    setBatchRegisterResults({ success, failed });
  };
  ```

### 2. 官方NFT发放功能

#### 2.1 单个发放流程

**用户操作路径**：管理员面板 → 单个发放标签 → 输入地址 → 选择NFT类型 → 点击发放

**官方NFT类型**：
- Token ID 1：📜 毕业证书（供应量10000）
- Token ID 2：🏅 纪念徽章（供应量10000）
- Token ID 3：🏆 荣誉证书（供应量1000，稀有）
- Token ID 4：⭐ 特殊奖项（供应量100，传说）

**完整数据流**：

```
1. 前置检查
   └→ frontend/app/admin/page.tsx
       └→ handleMint() 函数
           ├─ 检查地址是否已注册
           ├─ 检查是否已拥有该NFT
           └─ 通过则继续

2. 智能合约调用
   └→ writeContract()
       └→ mintOfficialNFT(address, tokenId)

3. 合约执行
   └→ contracts/GraduationNFT.sol
       └→ mintOfficialNFT() 函数 (第227-263行)
           ├─ 验证接收者已注册
           ├─ 验证tokenId有效性（1-4）
           ├─ 检查供应量未达上限
           ├─ **检查接收者是否已拥有（防重复发放）**
           ├─ 铸造NFT：_mint(to, tokenId, 1, "")
           ├─ 更新毕业生状态
           └─ 触发OfficialNFTMinted事件

4. 交易确认后
   └→ useEffect监听isSuccess
       └→ 同步到数据库
           └→ fetch('/api/db/mint', {...})

5. 数据库记录
   └→ frontend/app/api/db/mint/route.ts
       └→ 创建NFTRecord
       └→ 创建ActivityLog
       └→ 更新Statistics
```

  ```solidity
  function mintOfficialNFT(address to, uint256 tokenId) 
      external onlyRole(MINTER_ROLE) 
  {
      require(bytes(graduates[to].studentId).length > 0, "Graduate not registered");
      require(tokenId >= 1 && tokenId <= 4, "Invalid official NFT type");
      require(totalSupply(tokenId) < maxSupply[tokenId], "Max supply reached");
      require(balanceOf(to, tokenId) == 0, "Already owns this NFT");  // 防重复
      
      _mint(to, tokenId, 1, "");
      
      if (!graduates[to].hasClaimed) {
          graduates[to].hasClaimed = true;
          graduates[to].mintedAt = block.timestamp;
      }
      
      string memory nftType;
      if (tokenId == DIPLOMA) nftType = "Diploma";
      else if (tokenId == MEMORIAL_BADGE) nftType = "Memorial Badge";
      else if (tokenId == HONOR_CERTIFICATE) nftType = "Honor Certificate";
      else nftType = "Special Award";
      
      emit OfficialNFTMinted(to, tokenId, nftType);
  }
  ```

#### 2.2 批量发放流程

**用户操作路径**：管理员面板 → 批量发放标签 → 选择NFT类型 → 输入地址列表 → 点击批量发放

**完整数据流**：

```
1. 地址列表准备
   └→ 每行一个地址
       └→ 前端验证地址格式

2. 批量发放循环
   └→ handleBatchMint() 函数
       └→ 对每个地址：
           ├─ 检查是否已拥有该NFT
           ├─ 如已拥有则跳过（skipped++）
           ├─ 调用mintOfficialNFT()
           ├─ 等待确认
           └─ 同步到数据库

3. 智能合约执行
   └→ 可选两种方式：
       ├─ 方式1：循环调用单个mintOfficialNFT()（当前实现）
       └─ 方式2：调用batchMintOfficial()（批量优化）

4. 进度显示
   └→ BatchProgress组件
       ├─ 显示进度条
       └─ 显示结果：成功/失败/跳过
```

**关键代码位置**：

- **批量发放逻辑**：`frontend/app/admin/page.tsx` (第343-406行)
  ```typescript
  const handleBatchMint = async () => {
    if (selectedStudents.length === 0) return;
    
    setBatchMinting(true);
    let success = 0, failed = 0, skipped = 0;
    const tokenId = parseInt(batchMintTokenId);
    
    for (let i = 0; i < selectedStudents.length; i++) {
      const studentAddr = selectedStudents[i];
      setBatchMintProgress({ current: i + 1, total: selectedStudents.length });

        // 检查是否已拥有

          functionName: 'balanceOf',
          args: [studentAddr, BigInt(tokenId)],
        });

          skipped++;
          continue;
        }
        
        // 发放
        const hash = await walletClient.writeContract({
          functionName: 'mintOfficialNFT',
          args: [studentAddr, BigInt(tokenId)],
        });
        
        await publicClient.waitForTransactionReceipt({ hash });
        success++;
      } catch (error) {
        failed++;
      }
    }
    
    setBatchMintResults({ success, failed, skipped });
  };
  ```

- **合约批量发放函数**：`contracts/GraduationNFT.sol` (第265-302行)
  ```solidity
  function batchMintOfficial(
      address[] memory recipients,
      uint256 tokenId
  ) external onlyRole(MINTER_ROLE) {
      require(tokenId >= 1 && tokenId <= 4, "Invalid official NFT type");
      require(totalSupply(tokenId) + recipients.length <= maxSupply[tokenId], 
              "Would exceed max supply");
      
      for (uint256 i = 0; i < recipients.length; i++) {
          // 跳过未注册的学生
          if (bytes(graduates[recipients[i]].studentId).length == 0) continue;
          
          // 跳过已拥有该NFT的学生（防重复）
          if (balanceOf(recipients[i], tokenId) > 0) continue;
          
          _mint(recipients[i], tokenId, 1, "");
          
          if (!graduates[recipients[i]].hasClaimed) {
              graduates[recipients[i]].hasClaimed = true;
              graduates[recipients[i]].mintedAt = block.timestamp;
          }
      }
      
      emit BatchMinted(recipients, tokenId);
  }
  ```

### 3. 个人NFT铸造功能（盲盒机制）

#### 3.1 铸造流程详解

**用户操作路径**：铸造页面 → 连接钱包 → 点击"铸造唯一NFT"

**盲盒机制核心**：
- 稀有度随机生成
- 保底系统（20次保底）
- 每日限制（5次/天）
- 唯一颜色和编号

**完整数据流**：

```
1. 前端准备
   └→ frontend/app/mint/page.tsx
       └→ handleMint() 函数
           ├─ 检查钱包连接
           ├─ 检查是否已注册
           ├─ 检查今日剩余次数
           └─ 调用合约

2. 智能合约执行
   └→ contracts/GraduationNFT.sol
       └→ mintPersonalNFT() 函数 (第318-406行)
           │
           ├─ 【步骤1】验证用户资格
           │   └→ require(已注册为毕业生)
           │
           ├─ 【步骤2】检查每日限制
           │   ├→ 计算今天的日期（天数）
           │   ├→ 如果是新的一天，重置计数
           │   └→ require(今日铸造次数 < 5)
           │
           ├─ 【步骤3】生成稀有度（核心逻辑）
           │   ├→ pityCounter++
           │   │
           │   ├→ if (pityCounter >= 20)  // 保底触发
           │   │   ├─ 生成随机数(0-100)
           │   │   ├─ if (random < 20) → 传说 (20%)
           │   │   └─ else → 稀有 (80%)
           │   │   └─ pityCounter = 0  // 重置
           │   │
           │   └→ else  // 正常概率
           │       ├─ 生成随机数(0-1000)
           │       ├─ if (random < 50) → 传说 (5%)
           │       │   └─ pityCounter = 0
           │       ├─ else if (random < 300) → 稀有 (25%)
           │       │   └─ pityCounter = 0
           │       └─ else → 普通 (70%)
           │           └─ pityCounter不变（继续累积）
           │
           ├─ 【步骤4】生成唯一颜色种子
           │   └→ keccak256(timestamp + prevrandao + sender + tokenId + total)
           │
           ├─ 【步骤5】分配唯一Token ID
           │   ├→ tokenId = nextPersonalTokenId
           │   ├→ nextPersonalTokenId++
           │   └→ totalPersonalMinted++
           │
           ├─ 【步骤6】保存NFT元数据
           │   └→ personalNFTs[tokenId] = PersonalNFTData{...}
           │
           ├─ 【步骤7】铸造NFT
           │   ├→ _mint(msg.sender, tokenId, 1, "")
           │   ├→ selfMintCount++
           │   └→ dailyMintCount++
           │
           └─ 【步骤8】触发事件
               └→ emit PersonalNFTMinted(...)

3. 前端监听事件
   └→ useEffect监听isSuccess
       └→ 解析交易收据
           └→ 从logs中提取事件数据
               ├─ tokenId
               ├─ rarity
               ├─ colorSeed
               └─ mintNumber

4. 数据同步
   ├→ 同步到数据库
   │   └→ fetch('/api/db/mint')
   │
   └→ 上传到IPFS
       └→ fetch('/api/ipfs/upload-personal')
           ├─ 生成SVG图片
           ├─ 上传SVG到Pinata
           ├─ 生成JSON元数据
           └─ 上传JSON到Pinata

5. 显示结果
   └→ 更新UI
       ├─ 显示NFT图片
       ├─ 显示稀有度
       ├─ 显示铸造编号
       └─ 显示Toast提示
```

**随机数生成详解**：

```solidity
function _generateRandom(uint256 max) private returns (uint256) {
    _randomNonce++;  // 递增nonce
    return uint256(keccak256(abi.encodePacked(
        block.timestamp,    // 当前区块时间
        block.prevrandao,   // 前一个区块的随机数（PoS）
        msg.sender,         // 调用者地址
        _randomNonce        // nonce值
    ))) % max;
}
```

  ```solidity
      // 验证资格
      require(bytes(graduates[msg.sender].studentId).length > 0, "Not registered");
      
      // 检查每日限制
      uint256 today = block.timestamp / 1 days;
      if (lastMintDay[msg.sender] != today) {
          dailyMintCount[msg.sender] = 0;
          lastMintDay[msg.sender] = today;
      }

      // 生成稀有度
      graduates[msg.sender].pityCounter++;
      Rarity rarity;
      
      if (graduates[msg.sender].pityCounter >= PITY_COUNT) {
          // 保底触发
          uint256 pityRandom = _generateRandom(100);
          rarity = (pityRandom < 20) ? Rarity.LEGENDARY : Rarity.RARE;
          graduates[msg.sender].pityCounter = 0;
      } else {
          // 正常概率
          uint256 random = _generateRandom(1000);
          if (random < LEGENDARY_RATE) {
              rarity = Rarity.LEGENDARY;
              graduates[msg.sender].pityCounter = 0;
          } else if (random < LEGENDARY_RATE + RARE_RATE) {
              rarity = Rarity.RARE;
              graduates[msg.sender].pityCounter = 0;
          } else {
              rarity = Rarity.COMMON;
          }
      }
      
      // 生成唯一颜色种子
      uint256 colorSeed = uint256(keccak256(abi.encodePacked(
          block.timestamp, block.prevrandao, msg.sender, 
          nextPersonalTokenId, totalPersonalMinted
      )));
      
      // 分配Token ID
      uint256 tokenId = nextPersonalTokenId;
      nextPersonalTokenId++;
      totalPersonalMinted++;

          rarity: rarity,
  ```

```typescript
async function uploadFileToIPFS(svgContent: string, fileName: string): Promise<string> {
  const FormData = (await import('form-data')).default;
  const axios = (await import('axios')).default;
  
  const formData = new FormData();
  
  // 将SVG作为Buffer添加
  formData.append('file', Buffer.from(svgContent, 'utf-8'), {
    filename: fileName,
    contentType: 'image/svg+xml',
  });
  
  formData.append('pinataMetadata', JSON.stringify({ name: fileName }));
  
  const headers = {
    ...getPinataHeaders(),
    ...formData.getHeaders(),
  };
  
  // 使用axios发送请求（比fetch更可靠）
  const response = await axios.post(PINATA_PIN_FILE_URL, formData, {
    headers: headers,
    maxBodyLength: Infinity,
  });
  
  return response.data.IpfsHash;  // 返回CID
}
```

```
├──────────────────────────────────────────────────────────┤

                        ↕ Pinata API
```
